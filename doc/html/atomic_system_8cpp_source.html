182 int atomicSystem::setupGeometry()
183 {
184 
185 //! \brief set up the computational geometry - the grid - see T.Arias' lecture for further explanation
186 
187 using namespace arma;
188 
189 string lString="$\\mathbf{S}: 3 \\times 1$ number of sample points along each lattice vector ";
190 _ltX->newLine(lString);
191 
192 lString="$ \\Bigg( \\begin{array}{ccc}"+std::to_string(_Pa.S[0])+" \& 0 \& 0 \\\\ ";
193 lString+=" 0 \& "+ std::to_string(_Pa.S[1]) +" \& 0 \\\\ ";
194 lString+=" 0 \& 0 \& " + std::to_string(_Pa.S[2]) + "\\end{array} \\Bigg) $ ";
195 _ltX->newLine(lString);
196 
197 Col<double> ms(_Pa.prodS);
198 for(unsigned int i=0;i<ms.n_elem;i++)
199 { ms[i]=i; }
200 
201 Col<double> m1(_Pa.prodS);
202 Col<double> m2(_Pa.prodS);
203 Col<double> m3(_Pa.prodS);
204 
205 for(unsigned int i=0;i<ms.n_elem;i++)
206 {
207 m1[i]=std::fmod(ms(i),_Pa.S[0]);
208 m2[i]=std::floor(std::fmod(ms(i)/_Pa.S[0],_Pa.S[1]));
209 m3[i]=std::floor(std::fmod(ms(i)/(_Pa.S[0]*_Pa.S[1]),_Pa.S[2]));
210 }
211 
212 
213 Mat<double> M(_Pa.prodS,3);
214 M.fill(0);
215 M.col(0)=m1;
216 M.col(1)=m2;
217 M.col(2)=m3;
218 
219 Col<double> n1(_Pa.prodS);
220 Col<double> n2(_Pa.prodS);
221 Col<double> n3(_Pa.prodS);
222 
223 
224 for(unsigned int j=0;j<m1.n_rows;j++)
225  {
226  //value = (expression) ? (if true) : (if false);
227  n1[j] = (m1[j]<=(_Pa.S[0]/2)) ? (m1[j]) : (-_Pa.S[0]+m1[j]);
228  n2[j] = (m2[j]<=(_Pa.S[1]/2)) ? (m2[j]) : (-_Pa.S[1]+m2[j]);
229  n3[j] = (m3[j]<=(_Pa.S[2]/2)) ? (m3[j]) : (-_Pa.S[2]+m3[j]);
230 
231  }
232 
233 Mat<double> N(_Pa.prodS,3);
234 N.fill(0);
235 N.col(0)=n1;
236 N.col(1)=n2;
237 N.col(2)=n3;
238 
239 lString=" $\\mathbf{R}: 3 \\times 3$ matrix, each of whose columns is a lattice vector $ R_{k},\\mathbf{R}=\[R_{1},R_{2}, R_{3}\] $ ";
240 _ltX->newLine(lString);
241 lString="$ \\Bigg( \\begin{array}{ccc}"+std::to_string(_Pa.R[0,0])+" \& 0 \& 0 \\\\ ";
242 lString+=" 0 \& "+ std::to_string(_Pa.R[1,1]) +" \& 0 \\\\ ";
243 lString+=" 0 \& 0 \& " + std::to_string(_Pa.R[2,2]) + "\\end{array} \\Bigg) $ ";
244 _ltX->newLine(lString);
245 
246 Mat<double> diagS = diagmat(_Pa.S);
247 Mat<double> idiagS = inv((mat)diagS);
248 
249 _ltX->emptyLine();
250 lString=" $\\mathbf{r}: S_{k} \\times 3$ matrix, matrix, each of whose rows contains the real-space coordinates of a sample point"
251 " standard order $\\mathbf{r} ≡ \\mathbf{M} Diag(\\vec(S))^{-1}\\mathbf{R}^{T}$ ";
252 _ltX->newLine(lString);
253 
254 _r = M*idiagS*_Pa.R.t();
255 
256 verbosity(_Pa,"compute center of cell",2,__FILE__,__LINE__);
257 
258 for(int j=0;j<_Pa.prodS;j++)
259  {
260  _mat_center_of_cell.row(j)=(arma::sum(_Pa.R,1)/2).t();
261  }
262 
263 _dr=arma::sqrt(arma::sum((_r-_mat_center_of_cell)%(_r-_mat_center_of_cell),1));
264 
265 _ltX->emptyLine();
266 _ltX->newLine(" The G operator: ");
267 _ltX->newLine(" components of reciprocal lattice vector: ");
268 _ltX->newLine(" $\\Pi_{k} S_{k} \\times 3 matrix$ ");
269 _ltX->newLine(" $ \\mathbf{G} = 2 \\pi \\mathbf{N} \\mathbf{R}^{-1}$ ");
270 
271 _G=2*arma::datum::pi*N*inv(_Pa.R);
272 
273 _ltX->emptyLine();
274 _ltX->newLine(" The G2 operator: ");
275 _ltX->newLine(" components of square magnitude of the vector in the corresponding row of G: ");
276 _ltX->newLine(" $\\Pi_{k} S_{k} \\times 1 matrix$ ");
277 _ltX->newLine(" $ G2 = G^{2}$ ");
278 
279 _G2=arma::sum(_G%_G,1);
280 
281 verbosity(_Pa,"now compute the structure factor",2,__FILE__,__LINE__);
282 
283 verbosity(_Pa,"Compute structure factor using G: "+std::to_string(_G.n_rows)+" x "+std::to_string(_G.n_cols),2,__FILE__,__LINE__);
284 verbosity(_Pa,"Compute structure factor using X: "+std::to_string(_X.n_rows)+" x "+std::to_string(_X.n_cols),2,__FILE__,__LINE__);
285 
286 myFunctions::cassert(_X.n_elem>0,ISCRITICAL,"no atomic coordinates read so far - execute readAtomicCoordinates before setupGeometry!",__FILE__,__LINE__);
287 
288 arma::mat tmpmat(-_G*_X);
289 
290 verbosity(_Pa,"Now compute exponent of temporary matrix: "+std::to_string(tmpmat.n_rows)+" x "+std::to_string(tmpmat.n_cols),2,__FILE__,__LINE__);
291 
292 arma::cx_mat exponent(tmpmat.n_rows,tmpmat.n_cols,fill::zeros);
293 exponent.set_imag(tmpmat);
294 
295 _ltX->emptyLine();
296 _ltX->newLine(" The Sf operator: ");
297 _ltX->newLine(" structure factor associated with the wave vector of the corresponding column of G: ");
298 _ltX->newLine(" $ \\Pi_{k} S_{k} \\times 1$ column vector (one column matrix) ");
299 _ltX->newLine(" $ Sf(\\vec(G)) = \\Sigma_{I} \\exp(-i \\vec(G) \\vec(X)_{I}) $ ");
300 
301 verbosity(_Pa,"Sf is a sum over all atoms I of exp(G*X_{I})!",2,__FILE__,__LINE__);
302 verbosity(_Pa,"Sf is a column vector Sx1!",2,__FILE__,__LINE__);
303 
304 _Sf=sum(exp(exponent),1);
305 
306 verbosity(_Pa,"====================================",0,__FILE__,__LINE__);
307 verbosity(_Pa,"atomic: setup finished",0,__FILE__,__LINE__);
308 verbosity(_Pa,"====================================",0,__FILE__,__LINE__);
309 
310 
311 return 0;
